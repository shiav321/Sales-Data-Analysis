Sample Query
SQL in Action
-- Top 5 Products by Revenue SELECT p.name AS product_name, SUM(oi.quantity * oi.unit_price) AS total_revenue, COUNT(oi.item_id) AS total_units_sold FROM order_items oi JOIN products p ON oi.product_id = p.product_id GROUP BY p.name ORDER BY total_revenue DESC LIMIT 5; -- Monthly Revenue Trend SELECT DATE_FORMAT(order_date, '%Y-%m') AS month, SUM(total_amount) AS monthly_revenue FROM orders GROUP BY month ORDER BY month;
